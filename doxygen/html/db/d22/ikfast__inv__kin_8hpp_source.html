65 IKSolutions IKFastInvKin::calcInvKin(const Eigen::Isometry3d& pose, const Eigen::Ref<const Eigen::VectorXd>& seed) const
66 {
67  // Convert to ikfast data type
68  Eigen::Transform<IkReal, 3, Eigen::Isometry> ikfast_tcp = pose.cast<IkReal>();
69 
70  // Decompose
71  const Eigen::Matrix<IkReal, 3, 1> translation = ikfast_tcp.translation();
72 
73  // Note the row major ordering here: IkFast expects the matrix in r00, r01, r02, ..., r11, r12, r13
74  // ordering
75  const Eigen::Matrix<IkReal, 3, 3, Eigen::RowMajor> rotation = ikfast_tcp.rotation();
76 
77  // Call IK (TODO: Make a better solution list class? One that uses vector instead of list)
78  ikfast::IkSolutionList<IkReal> ikfast_solution_set;
79  ComputeIk(translation.data(), rotation.data(), nullptr, ikfast_solution_set);
80 
81  // Unpack the solutions into the output vector
82  const auto n_sols = ikfast_solution_set.GetNumSolutions();
83  int ikfast_dof = numJoints();
84 
85  std::vector<IkReal> ikfast_output;
86  ikfast_output.resize(n_sols * ikfast_dof);
87 
88  for (std::size_t i = 0; i < n_sols; ++i)
89  {
90  // This actually walks the list EVERY time from the start of i.
91  const auto& sol = ikfast_solution_set.GetSolution(i);
92  auto* out = ikfast_output.data() + i * ikfast_dof;
93  sol.GetSolution(out, nullptr);
94  }
95 
96  std::vector<double> sols;
97  sols.insert(end(sols), std::make_move_iterator(ikfast_output.begin()), std::make_move_iterator(ikfast_output.end()));
98 
99  // Check the output
100  int num_sol = sols.size() / ikfast_dof;
101  IKSolutions solution_set;
102  solution_set.reserve(sols.size());
103  for (int i = 0; i < num_sol; i++)
104  {
105  Eigen::Map<Eigen::VectorXd> eigen_sol(sols.data() + ikfast_dof * i, static_cast<Eigen::Index>(ikfast_dof));
106  if (eigen_sol.array().allFinite())
107  {
108  harmonizeTowardZero<double>(eigen_sol); // Modifies 'sol' in place
109 
110  // Add solution
111  if (tesseract_common::satisfiesPositionLimits(eigen_sol, limits_.joint_limits))
112  solution_set.push_back(eigen_sol);
113  }
114  }
115 
116  return solution_set;
117 }
