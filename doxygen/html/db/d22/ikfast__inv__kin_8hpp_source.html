73 inline IKSolutions IKFastInvKin::calcInvKin(const tesseract_common::TransformMap& tip_link_poses,
74  const Eigen::Ref<const Eigen::VectorXd>& /*seed*/) const
75 {
76  assert(tip_link_poses.size() == 1);
77  assert(tip_link_poses.find(tip_link_name_) != tip_link_poses.end());
78 
79  const Eigen::Isometry3d& pose = tip_link_poses.at(tip_link_name_);
80 
81  // Convert to ikfast data type
82  Eigen::Transform<IkReal, 3, Eigen::Isometry> ikfast_tcp = pose.cast<IkReal>();
83 
84  // Decompose
85  const Eigen::Matrix<IkReal, 3, 1> translation = ikfast_tcp.translation();
86 
87  // Note the row major ordering here: IkFast expects the matrix in r00, r01, r02, ..., r11, r12, r13
88  // ordering
89  const Eigen::Matrix<IkReal, 3, 3, Eigen::RowMajor> rotation = ikfast_tcp.rotation();
90 
91  // Call IK (TODO: Make a better solution list class? One that uses vector instead of list)
92  ikfast::IkSolutionList<IkReal> ikfast_solution_set;
93  ComputeIk(translation.data(), rotation.data(), nullptr, ikfast_solution_set);
94 
95  // Unpack the solutions into the output vector
96  const std::size_t n_sols = ikfast_solution_set.GetNumSolutions();
97  const auto ikfast_dof = static_cast<std::size_t>(numJoints());
98 
99  std::vector<IkReal> ikfast_output;
100  ikfast_output.resize(n_sols * ikfast_dof);
101 
102  for (std::size_t i = 0; i < n_sols; ++i)
103  {
104  // This actually walks the list EVERY time from the start of i.
105  const auto& sol = ikfast_solution_set.GetSolution(i);
106  auto* out = ikfast_output.data() + i * ikfast_dof;
107  sol.GetSolution(out, nullptr);
108  }
109 
110  std::vector<double> sols;
111  sols.insert(end(sols), std::make_move_iterator(ikfast_output.begin()), std::make_move_iterator(ikfast_output.end()));
112 
113  // Check the output
114  int num_sol = static_cast<int>(sols.size() / ikfast_dof);
115  IKSolutions solution_set;
116  solution_set.reserve(sols.size());
117  for (int i = 0; i < num_sol; i++)
118  {
119  Eigen::Map<Eigen::VectorXd> eigen_sol(sols.data() + static_cast<Eigen::Index>(ikfast_dof) * i,
120  static_cast<Eigen::Index>(ikfast_dof));
121  if (eigen_sol.array().allFinite())
122  {
123  harmonizeTowardZero<double>(eigen_sol); // Modifies 'sol' in place
124  solution_set.push_back(eigen_sol);
125  }
126  }
127 
128  return solution_set;
129 }
