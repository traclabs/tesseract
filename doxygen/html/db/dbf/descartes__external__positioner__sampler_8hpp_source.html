72 template <typename FloatType>
73 bool DescartesExternalPositionerSampler<FloatType>::sample(std::vector<FloatType>& solution_set)
74 {
75  double distance = std::numeric_limits<double>::min();
76  auto num_joints = static_cast<int>(positioner_kinematics_->numJoints());
77 
78  // For the kinematics object to be sampled we need to create the joint values at the sampling resolution
79  // The sampled joints results are stored in dof_range[joint index] to be used by the nested_ik function
80  std::vector<Eigen::VectorXd> dof_range;
81  dof_range.reserve(static_cast<std::size_t>(num_joints));
82  for (long dof = 0; dof < num_joints; ++dof)
83  {
84  // given the sampling resolution for the joint calculate the number of samples such that the resolution is not
85  // exceeded.
86  long cnt = static_cast<long>(std::ceil(std::abs(positioner_limits_(dof, 1) - positioner_limits_(dof, 0)) /
87  positioner_sample_resolution_(dof))) +
88  1;
89  dof_range.push_back(Eigen::VectorXd::LinSpaced(cnt, positioner_limits_(dof, 0), positioner_limits_(dof, 1)));
90  }
91 
92  tesseract_common::VectorIsometry3d target_poses = target_pose_sampler_(target_pose_);
93  for (const auto& sp : target_poses)
94  {
95  // Tool pose subtract robot tcp
96  Eigen::Isometry3d target_pose = sp * robot_tcp_.inverse();
97  Eigen::Matrix<FloatType, Eigen::Dynamic, 1> positioner_pose(num_joints);
98  nested_ik(solution_set, 0, dof_range, target_pose, positioner_pose, false, distance);
99  }
100 
101  if (solution_set.empty() && allow_collision_)
102  getBestSolution(solution_set, target_poses, dof_range);
103 
104  return !solution_set.empty();
105 }

139 template <typename FloatType>
140 bool DescartesExternalPositionerSampler<FloatType>::ikAt(
141  std::vector<FloatType>& solution_set,
142  const Eigen::Isometry3d& target_pose,
143  const Eigen::Ref<const Eigen::Matrix<FloatType, Eigen::Dynamic, 1>>& positioner_pose,
144  bool get_best_solution,
145  double& distance)
146 {
147  Eigen::Isometry3d positioner_tf;
148  if (!positioner_kinematics_->calcFwdKin(positioner_tf, positioner_pose.template cast<double>()))
149  return false;
150 
151  Eigen::Isometry3d robot_target_pose =
152  world_to_robot_base_.inverse() * (world_to_positioner_base_ * positioner_tf * target_pose);
153  if (robot_target_pose.translation().norm() > robot_reach_)
154  return false;
155 
156  Eigen::VectorXd robot_solution_set;
157  auto robot_dof = static_cast<int>(robot_kinematics_->numJoints());
158  if (!robot_kinematics_->calcInvKin(robot_solution_set, robot_target_pose, ik_seed_))
159  return false;
160 
161  long num_sols = robot_solution_set.size() / robot_dof;
162  for (long i = 0; i < num_sols; i++)
163  {
164  double* sol = robot_solution_set.data() + robot_dof * i;
165 
166  std::vector<FloatType> full_sol;
167  full_sol.insert(end(full_sol), positioner_pose.data(), positioner_pose.data() + positioner_pose.size());
168  full_sol.insert(end(full_sol), std::make_move_iterator(sol), std::make_move_iterator(sol + robot_dof));
169 
170  if ((is_valid_ != nullptr) && !is_valid_(Eigen::Map<Eigen::Matrix<FloatType, Eigen::Dynamic, 1>>(
171  full_sol.data(), static_cast<long>(full_sol.size()))))
172  continue;
173 
174  if (!get_best_solution)
175  {
176  if (isCollisionFree(full_sol.data()))
177  solution_set.insert(
178  end(solution_set), std::make_move_iterator(full_sol.begin()), std::make_move_iterator(full_sol.end()));
179  }
180  else
181  {
182  double cur_distance = collision_->distance(full_sol.data(), full_sol.size());
183  if (cur_distance > distance)
184  {
185  distance = cur_distance;
186  solution_set.insert(
187  begin(solution_set), std::make_move_iterator(full_sol.begin()), std::make_move_iterator(full_sol.end()));
188  }
189  }
190  }
191 
192  return !solution_set.empty();
193 }
