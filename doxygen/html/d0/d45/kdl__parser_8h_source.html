60 inline KDL::Frame convert(const Eigen::Isometry3d& transform)
61 {
62  KDL::Frame frame;
63  frame.Identity();
64 
65  for (int i = 0; i < 3; ++i)
66  frame.p[i] = transform(i, 3);
67 
68  for (int i = 0; i < 9; ++i)
69  frame.M.data[i] = transform(i / 3, i % 3);
70 
71  return frame;
72 }

86 inline KDL::Joint convert(const Joint::ConstPtr& joint)
87 {
88  KDL::Frame parent_joint = convert(joint->parent_to_joint_origin_transform);
89  const std::string& name = joint->getName();
90 
91  switch (joint->type)
92  {
93  case JointType::FIXED:
94  {
95  return KDL::Joint(name, KDL::Joint::None);
96  }
97  case JointType::REVOLUTE:
98  {
99  KDL::Vector axis = convert(joint->axis);
100  return KDL::Joint(name, parent_joint.p, parent_joint.M * axis, KDL::Joint::RotAxis);
101  }
102  case JointType::CONTINUOUS:
103  {
104  KDL::Vector axis = convert(joint->axis);
105  return KDL::Joint(name, parent_joint.p, parent_joint.M * axis, KDL::Joint::RotAxis);
106  }
107  case JointType::PRISMATIC:
108  {
109  KDL::Vector axis = convert(joint->axis);
110  return KDL::Joint(name, parent_joint.p, parent_joint.M * axis, KDL::Joint::TransAxis);
111  }
112  default:
113  {
114  CONSOLE_BRIDGE_logWarn("Converting unknown joint type of joint '%s' into a fixed joint", name.c_str());
115  return KDL::Joint(name, KDL::Joint::None);
116  }
117  }

125 inline KDL::RigidBodyInertia convert(const Inertial::ConstPtr& inertial)
126 {
127  KDL::Frame origin = convert(inertial->origin);
128 
129  // the mass is frame independent
130  double kdl_mass = inertial->mass;
131 
132  // kdl and urdf both specify the com position in the reference frame of the link
133  KDL::Vector kdl_com = origin.p;
134 
135  // kdl specifies the inertia matrix in the reference frame of the link,
136  // while the urdf specifies the inertia matrix in the inertia reference frame
137  KDL::RotationalInertia urdf_inertia =
138  KDL::RotationalInertia(inertial->ixx, inertial->iyy, inertial->izz, inertial->ixy, inertial->ixz, inertial->iyz);
139 
140  // Rotation operators are not defined for rotational inertia,
141  // so we use the RigidBodyInertia operators (with com = 0) as a workaround
142  KDL::RigidBodyInertia kdl_inertia_wrt_com_workaround =
143  origin.M * KDL::RigidBodyInertia(0, KDL::Vector::Zero(), urdf_inertia);
144 
145  // Note that the RigidBodyInertia constructor takes the 3d inertia wrt the com
146  // while the getRotationalInertia method returns the 3d inertia wrt the frame origin
147  // (but having com = Vector::Zero() in kdl_inertia_wrt_com_workaround they match)
148  KDL::RotationalInertia kdl_inertia_wrt_com = kdl_inertia_wrt_com_workaround.getRotationalInertia();
149 
150  return KDL::RigidBodyInertia(kdl_mass, kdl_com, kdl_inertia_wrt_com);
151 }

161  template <class u, class g>
162  void discover_vertex(u vertex, g graph)
163  {
164  const Link::ConstPtr& link = boost::get(boost::vertex_link, graph)[vertex];
165 
166  // constructs the optional inertia
167  KDL::RigidBodyInertia inert(0);
168  if (link->inertial)
169  inert = convert(link->inertial);
170 
171  // Get incomming edges
172  auto num_in_edges = static_cast<int>(boost::in_degree(vertex, graph));
173  if (num_in_edges == 0) // The root of the tree will have not incoming edges
174  return;
175 
176  boost::graph_traits<Graph>::in_edge_iterator ei, ei_end;
177  boost::tie(ei, ei_end) = boost::in_edges(vertex, graph);
178  SceneGraph::Edge e = *ei;
179  const Joint::ConstPtr& parent_joint = boost::get(boost::edge_joint, graph)[e];
180  KDL::Joint kdl_jnt = convert(parent_joint);
181 
182  // construct the kdl segment
183  KDL::Segment sgm(link->getName(), kdl_jnt, convert(parent_joint->parent_to_joint_origin_transform), inert);
184 
185  // add segment to tree
186  tree_.addSegment(sgm, parent_joint->parent_link_name);
187  }
188 
189 protected:
190  KDL::Tree& tree_;
191 };

200 inline bool parseSceneGraph(const SceneGraph& scene_graph, KDL::Tree& tree)
201 {
202  assert(scene_graph.isTree());
203 
204  const std::string& root_name = scene_graph.getRoot();
205  const Link::ConstPtr& root_link = scene_graph.getLink(root_name);
206 
207  tree = KDL::Tree(root_name);
208 
209  // warn if root link has inertia. KDL does not support this
210  if (root_link->inertial)
211  {
212  CONSOLE_BRIDGE_logWarn("The root link %s has an inertia specified in the URDF, but KDL does not "
213  "support a root link with an inertia. As a workaround, you can add an extra "
214  "dummy link to your URDF.",
215  root_name.c_str());
216  }
217 
218  kdl_tree_builder builder(tree);
219 
220  std::map<SceneGraph::Vertex, size_t> index_map;
221  boost::associative_property_map<std::map<SceneGraph::Vertex, size_t>> prop_index_map(index_map);
222 
223  int c = 0;
224  Graph::vertex_iterator i, iend;
225  for (boost::tie(i, iend) = boost::vertices(scene_graph); i != iend; ++i, ++c)
226  boost::put(prop_index_map, *i, c);
227 
228  boost::depth_first_search(
229  static_cast<const Graph&>(scene_graph),
230  boost::visitor(builder).root_vertex(scene_graph.getVertex(root_name)).vertex_index_map(prop_index_map));
231  return true;
232 }
