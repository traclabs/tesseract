101  {
102  std::vector<std::string> tokens;
103  boost::split(tokens, scale_string, boost::is_any_of(" "), boost::token_compress_on);
104  if (tokens.size() != 3 || !tesseract_common::isNumeric(tokens))
105  return std::make_shared<SC>(ConvexMeshStatusCategory::ErrorParsingAttributeScale, status_cat);
106 
107  double sx{ 0 }, sy{ 0 }, sz{ 0 };
108  // No need to check return values because the tokens are verified above
109  tesseract_common::toNumeric<double>(tokens[0], sx);
110  tesseract_common::toNumeric<double>(tokens[1], sy);
111  tesseract_common::toNumeric<double>(tokens[2], sz);
112 
113  scale = Eigen::Vector3d(sx, sy, sz);
114  }
115

117  xml_element->QueryBoolAttribute("convert", &convert);
118

121  locator->locateResource(filename), scale, true, true, true, true, true);
122  else
123  {

127  locator->locateResource(filename), scale, false, false);
