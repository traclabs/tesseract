106 tesseract_common::StatusCode DescartesMotionPlanner<FloatType>::solve(PlannerResponse& response,
107  PostPlanCheckType check_type,
108  const bool verbose)
109 {
110  tesseract_common::StatusCode config_status = isConfigured();
111  if (!config_status)
112  {
113  response.status = config_status;
114  CONSOLE_BRIDGE_logError("Planner %s is not configured", name_.c_str());
115  return config_status;
116  }
117 
118  auto tStart = boost::posix_time::second_clock::local_time();
119 
120  const auto dof = config_->joint_names.size();
121  descartes_light::Solver<FloatType> graph_builder(dof);
122  if (!graph_builder.build(
123  config_->samplers, config_->timing_constraint, config_->edge_evaluator, config_->num_threads))
124  {
125  CONSOLE_BRIDGE_logError("Failed to build vertices");
126  for (const auto& i : graph_builder.getFailedVertices())
127  response.failed_waypoints.push_back(config_->waypoints[i]);
128 
129  // Copy the waypoint if it is not already in the failed waypoints list
130  std::copy_if(config_->waypoints.begin(),
131  config_->waypoints.end(),
132  std::back_inserter(response.succeeded_waypoints),
133  [&response](const Waypoint::ConstPtr wp) {
134  return std::find(response.failed_waypoints.begin(), response.failed_waypoints.end(), wp) ==
135  response.failed_waypoints.end();
136  });
137 
138  response.status =
139  tesseract_common::StatusCode(DescartesMotionPlannerStatusCategory::ErrorFailedToBuildGraph, status_category_);
140  return response.status;
141  }
142  // No failed waypoints
143  response.succeeded_waypoints = config_->waypoints;
144  response.failed_waypoints.clear();
145 
146  // Search for edges
147  std::vector<FloatType> solution;
148  if (!graph_builder.search(solution))
149  {
150  CONSOLE_BRIDGE_logError("Search for graph completion failed");
151  response.status = tesseract_common::StatusCode(DescartesMotionPlannerStatusCategory::ErrorFailedToFindValidSolution,
152  status_category_);
153  return response.status;
154  }
155 
156  response.joint_trajectory.joint_names = config_->joint_names;
157  response.joint_trajectory.trajectory.resize(static_cast<long>(config_->waypoints.size()), static_cast<long>(dof));
158  for (size_t r = 0; r < config_->waypoints.size(); ++r)
159  for (size_t c = 0; c < dof; ++c)
160  response.joint_trajectory.trajectory(static_cast<long>(r), static_cast<long>(c)) = solution[(r * dof) + c];
161 
162  // Check and report collisions
163  validator_ =
164  std::make_shared<TrajectoryValidator>(config_->tesseract->getEnvironmentConst()->getContinuousContactManager(),
165  config_->tesseract->getEnvironmentConst()->getDiscreteContactManager(),
166  0.01,
167  verbose);
168 
169  bool valid = validator_->trajectoryValid(response.joint_trajectory.trajectory,
170  check_type,
171  *(config_->tesseract->getEnvironmentConst()->getStateSolver()),
172  response.joint_trajectory.joint_names);
173 
174  CONSOLE_BRIDGE_logInform("Descartes planning time: %.3f",
175  (boost::posix_time::second_clock::local_time() - tStart).seconds());
176 
177  if (!valid)
178  {
179  response.status = tesseract_common::StatusCode(
180  DescartesMotionPlannerStatusCategory::ErrorFoundValidSolutionInCollision, status_category_);
181  return response.status;
182  }
183 
184  CONSOLE_BRIDGE_logInform("Final trajectory is collision free");
185  response.status = tesseract_common::StatusCode(DescartesMotionPlannerStatusCategory::SolutionFound, status_category_);
186  return response.status;
