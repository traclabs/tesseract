165 inline bool parseSceneGraph(KDLChainData& results,
166  const tesseract_scene_graph::SceneGraph& scene_graph,
167  const std::vector<std::pair<std::string, std::string>>& chains)
168 {
169  if (!tesseract_scene_graph::parseSceneGraph(scene_graph, results.kdl_tree))
170  {
171  CONSOLE_BRIDGE_logError("Failed to parse KDL tree from Scene Graph");
172  return false;
173  }
174 
175  results.chains = chains;
176  results.base_name = chains.front().first;
177  for (const auto& chain : chains)
178  {
179  KDL::Chain sub_chain;
180  if (!results.kdl_tree.getChain(chain.first, chain.second, sub_chain))
181  {
182  CONSOLE_BRIDGE_logError(
183  "Failed to initialize KDL between links: '%s' and '%s'", chain.first.c_str(), chain.second.c_str());
184  return false;
185  }
186  results.robot_chain.addChain(sub_chain);
187  }
188  results.tip_name = chains.back().second;
189 
190  results.joint_list.resize(results.robot_chain.getNrOfJoints());
191  results.limits.joint_limits.resize(results.robot_chain.getNrOfJoints(), 2);
192  results.limits.velocity_limits.resize(results.robot_chain.getNrOfJoints());
193  results.limits.acceleration_limits.resize(results.robot_chain.getNrOfJoints());
194 
195  results.segment_index[results.base_name] = 0;
196  results.link_list.push_back(results.base_name);
197  results.active_link_list.clear();
198  bool found = false;
199  for (unsigned i = 0, j = 0; i < results.robot_chain.getNrOfSegments(); ++i)
200  {
201  const KDL::Segment& seg = results.robot_chain.getSegment(i);
202  const KDL::Joint& jnt = seg.getJoint();
203  results.link_list.push_back(seg.getName());
204 
205  // KDL segments does not contain the the base link in this list. When calling function that take segmentNr, like
206  // JntToCart to get the base link transform you would pass an index of zero and for subsequent links it is
207  // index + 1. This was determined through testing which is captured in this packages unit tests.
208  results.segment_index[seg.getName()] = static_cast<int>(i + 1);
209 
210  if (found)
211  results.active_link_list.push_back(seg.getName());
212 
213  if (jnt.getType() == KDL::Joint::None)
214  continue;
215 
216  if (!found)
217  {
218  found = true;
219  results.active_link_list.push_back(seg.getName());
220  }
221 
222  results.joint_list[j] = jnt.getName();
223  const tesseract_scene_graph::Joint::ConstPtr& joint = scene_graph.getJoint(jnt.getName());
224  results.limits.joint_limits(j, 0) = joint->limits->lower;
225  results.limits.joint_limits(j, 1) = joint->limits->upper;
226  results.limits.velocity_limits(j) = joint->limits->velocity;
227  results.limits.acceleration_limits(j) = joint->limits->acceleration;
228 
229  // Need to set limits for continuous joints. TODO: This may not be required
230  // by the optimization library but may be nice to have
231  if (joint->type == tesseract_scene_graph::JointType::CONTINUOUS &&
232  tesseract_common::almostEqualRelativeAndAbs(
233  results.limits.joint_limits(j, 0), results.limits.joint_limits(j, 1), 1e-5))
234  {
235  results.limits.joint_limits(j, 0) = -4 * M_PI;
236  results.limits.joint_limits(j, 1) = +4 * M_PI;
237  }
238  ++j;
239  }
240 
241  results.redundancy_indices.clear();
242  for (std::size_t i = 0; i < results.joint_list.size(); ++i)
243  {
244  const auto& joint = scene_graph.getJoint(results.joint_list[i]);
245  switch (joint->type)
246  {
247  case tesseract_scene_graph::JointType::REVOLUTE:
248  case tesseract_scene_graph::JointType::CONTINUOUS:
249  results.redundancy_indices.push_back(static_cast<Eigen::Index>(i));
250  break;
251  default:
252  break;
253  }
254  }
255 
256  return true;
257 }
