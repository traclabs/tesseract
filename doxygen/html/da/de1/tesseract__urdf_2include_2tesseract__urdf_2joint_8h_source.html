125 inline tesseract_common::StatusCode::Ptr parse(tesseract_scene_graph::Joint::Ptr& joint,
126  const tinyxml2::XMLElement* xml_element,
127  const int version)
128 {
129  joint = nullptr;
130  auto status_cat = std::make_shared<JointStatusCategory>();
131  using SC = tesseract_common::StatusCode;
132 
133  // get joint name
134  std::string joint_name;
135  if (tesseract_common::QueryStringAttribute(xml_element, "name", joint_name) != tinyxml2::XML_SUCCESS)
136  return std::make_shared<SC>(JointStatusCategory::ErrorMissingAttributeName, status_cat);
137 
138  status_cat = std::make_shared<JointStatusCategory>(joint_name);
139 
140  // create joint
141  auto j = std::make_shared<tesseract_scene_graph::Joint>(joint_name);
142 
143  // get joint origin
144  const tinyxml2::XMLElement* origin = xml_element->FirstChildElement("origin");
145  if (origin != nullptr)
146  {
147  auto status = parse(j->parent_to_joint_origin_transform, origin, version);
148  if (!(*status))
149  return std::make_shared<SC>(JointStatusCategory::ErrorParsingOrigin, status_cat, status);
150  }
151 
152  // get parent link
153  const tinyxml2::XMLElement* parent = xml_element->FirstChildElement("parent");
154  if (parent == nullptr)
155  return std::make_shared<SC>(JointStatusCategory::ErrorMissingParentElement, status_cat);
156 
157  if (tesseract_common::QueryStringAttribute(parent, "link", j->parent_link_name) != tinyxml2::XML_SUCCESS)
158  return std::make_shared<SC>(JointStatusCategory::ErrorParsingParentAttributeLink, status_cat);
159 
160  // get child link
161  const tinyxml2::XMLElement* child = xml_element->FirstChildElement("child");
162  if (child == nullptr)
163  return std::make_shared<SC>(JointStatusCategory::ErrorMissingChildElement, status_cat);
164 
165  if (tesseract_common::QueryStringAttribute(child, "link", j->child_link_name) != tinyxml2::XML_SUCCESS)
166  return std::make_shared<SC>(JointStatusCategory::ErrorParsingChildAttributeLink, status_cat);
167 
168  // get joint type
169  std::string joint_type;
170  if (tesseract_common::QueryStringAttribute(xml_element, "type", joint_type) != tinyxml2::XML_SUCCESS)
171  return std::make_shared<SC>(JointStatusCategory::ErrorMissingTypeElement, status_cat);
172 
173  if (joint_type == "planar")
174  j->type = tesseract_scene_graph::JointType::PLANAR;
175  else if (joint_type == "floating")
176  j->type = tesseract_scene_graph::JointType::FLOATING;
177  else if (joint_type == "revolute")
178  j->type = tesseract_scene_graph::JointType::REVOLUTE;
179  else if (joint_type == "continuous")
180  j->type = tesseract_scene_graph::JointType::CONTINUOUS;
181  else if (joint_type == "prismatic")
182  j->type = tesseract_scene_graph::JointType::PRISMATIC;
183  else if (joint_type == "fixed")
184  j->type = tesseract_scene_graph::JointType::FIXED;
185  else
186  return std::make_shared<SC>(JointStatusCategory::ErrorInvalidType, status_cat);
187 
188  // get joint axis
189  if (j->type != tesseract_scene_graph::JointType::FLOATING && j->type != tesseract_scene_graph::JointType::FIXED)
190  {
191  const tinyxml2::XMLElement* axis = xml_element->FirstChildElement("axis");
192  if (axis == nullptr)
193  {
194  j->axis = Eigen::Vector3d(1.0, 0.0, 0.0);
195  }
196  else
197  {
198  std::string axis_str;
199  if (tesseract_common::QueryStringAttribute(axis, "xyz", axis_str) != tinyxml2::XML_SUCCESS)
200  return std::make_shared<SC>(JointStatusCategory::ErrorParsingAxisAttributeXYZ, status_cat);
201 
202  std::vector<std::string> tokens;
203  boost::split(tokens, axis_str, boost::is_any_of(" "), boost::token_compress_on);
204  if (tokens.size() != 3 || !tesseract_common::isNumeric(tokens))
205  return std::make_shared<SC>(JointStatusCategory::ErrorParsingAxisAttributeXYZ, status_cat);
206 
207  double ax, ay, az;
208  // No need to check return values because the tokens are verified above
209  tesseract_common::toNumeric<double>(tokens[0], ax);
210  tesseract_common::toNumeric<double>(tokens[1], ay);
211  tesseract_common::toNumeric<double>(tokens[2], az);
212 
213  j->axis = Eigen::Vector3d(ax, ay, az);
214  }
215  }
216 
217  // get joint limits
218  if (j->type == tesseract_scene_graph::JointType::REVOLUTE || j->type == tesseract_scene_graph::JointType::PRISMATIC ||
219  j->type == tesseract_scene_graph::JointType::CONTINUOUS)
220  {
221  const tinyxml2::XMLElement* limits = xml_element->FirstChildElement("limit");
222  if (limits == nullptr && j->type != tesseract_scene_graph::JointType::CONTINUOUS)
223  {
224  return std::make_shared<SC>(JointStatusCategory::ErrorMissingLimitsElement, status_cat);
225  }
226 
227  if (limits == nullptr && j->type == tesseract_scene_graph::JointType::CONTINUOUS)
228  {
229  j->limits = std::make_shared<tesseract_scene_graph::JointLimits>();
230  }
231  else
232  {
233  auto status = parse(j->limits, limits, version);
234  if (!(*status))
235  return std::make_shared<SC>(JointStatusCategory::ErrorParsingLimits, status_cat, status);
236  }
237  }
238 
239  // get joint safety if exists
240  const tinyxml2::XMLElement* safety = xml_element->FirstChildElement("safety_controller");
241  if (safety != nullptr)
242  {
243  auto status = parse(j->safety, safety, version);
244  if (!(*status))
245  return std::make_shared<SC>(JointStatusCategory::ErrorParsingSafetyController, status_cat, status);
246  }
247 
248  // get joint calibration if exists
249  const tinyxml2::XMLElement* calibration = xml_element->FirstChildElement("calibration");
250  if (calibration != nullptr)
251  {
252  auto status = parse(j->calibration, calibration, version);
253  if (!(*status))
254  return std::make_shared<SC>(JointStatusCategory::ErrorParsingCalibration, status_cat, status);
255  }
256 
257  // get mimic joint if exists
258  const tinyxml2::XMLElement* mimic = xml_element->FirstChildElement("mimic");
259  if (mimic != nullptr)
260  {
261  auto status = parse(j->mimic, mimic, version);
262  if (!(*status))
263  return std::make_shared<SC>(JointStatusCategory::ErrorParsingMimic, status_cat, status);
264  }
265 
266  // get dynamics if exists
267  const tinyxml2::XMLElement* dynamics = xml_element->FirstChildElement("dynamics");
268  if (dynamics != nullptr)
269  {
270  auto status = parse(j->dynamics, dynamics, version);
271  if (!(*status))
272  return std::make_shared<SC>(JointStatusCategory::ErrorParsingDynamics, status_cat, status);
273  }
274 
275  joint = std::move(j);
276  return std::make_shared<SC>(JointStatusCategory::Success, status_cat);
277 }
