266  /* Create a clone of current joint, with a new name. Child link name and parent link name are unchanged.
267  * All underlying properties, such as dynamics, limits... are copied as well.*/
268  Joint clone(const std::string& name) const
269  {
270  Joint ret(name);
271  ret.axis = this->axis;
272  ret.child_link_name = this->child_link_name;
273  ret.parent_link_name = this->parent_link_name;
274  ret.parent_to_joint_origin_transform = this->parent_to_joint_origin_transform;
275  ret.type = this->type;
276  if (this->dynamics)
277  {
278  ret.dynamics = std::make_shared<JointDynamics>(*(this->dynamics));
279  }
280  if (this->limits)
281  {
282  ret.limits = std::make_shared<JointLimits>(*(this->limits));
283  }
284  if (this->safety)
285  {
286  ret.safety = std::make_shared<JointSafety>(*(this->safety));
287  }
288  if (this->calibration)
289  {
290  ret.calibration = std::make_shared<JointCalibration>(*(this->calibration));
291  }
292  if (this->mimic)
293  {
294  ret.mimic = std::make_shared<JointMimic>(*(this->mimic));
295  }
296  return ret;
297  }
