318  /* Create a clone of current joint, with a new name. Child link name and parent link name are unchanged.
319  * All underlying properties, such as dynamics, limits... are copied as well.*/
320  Joint clone(const std::string& name) const
321  {
322  Joint ret(name);
323  ret.axis = this->axis;
324  ret.child_link_name = this->child_link_name;
325  ret.parent_link_name = this->parent_link_name;
326  ret.parent_to_joint_origin_transform = this->parent_to_joint_origin_transform;
327  ret.type = this->type;
328  if (this->dynamics)
329  {
330  ret.dynamics = std::make_shared<JointDynamics>(*(this->dynamics));
331  }
332  if (this->limits)
333  {
334  ret.limits = std::make_shared<JointLimits>(*(this->limits));
335  }
336  if (this->safety)
337  {
338  ret.safety = std::make_shared<JointSafety>(*(this->safety));
339  }
340  if (this->calibration)
341  {
342  ret.calibration = std::make_shared<JointCalibration>(*(this->calibration));
343  }
344  if (this->mimic)
345  {
346  ret.mimic = std::make_shared<JointMimic>(*(this->mimic));
347  }
348  return ret;
349  }
