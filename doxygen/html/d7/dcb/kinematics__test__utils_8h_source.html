279 inline void runJacobianTest(tesseract_kinematics::ForwardKinematics& kin,
280  const Eigen::VectorXd& jvals,
281  const std::string& link_name,
282  const Eigen::Vector3d& link_point,
283  const Eigen::Isometry3d& change_base)
284 {
285  Eigen::MatrixXd jacobian, numerical_jacobian;
286  Eigen::Isometry3d pose;
287 
288  jacobian.resize(6, kin.numJoints());
289  if (link_name.empty())
290  {
291  pose = kin.calcFwdKin(jvals);
292  jacobian = kin.calcJacobian(jvals);
293  }
294  else
295  {
296  pose = kin.calcFwdKin(jvals, link_name);
297  jacobian = kin.calcJacobian(jvals, link_name);
298  }
299  tesseract_kinematics::jacobianChangeBase(jacobian, change_base);
300  tesseract_kinematics::jacobianChangeRefPoint(jacobian, (change_base * pose).linear() * link_point);
301 
302  numerical_jacobian.resize(6, kin.numJoints());
303  if (link_name.empty())
304  tesseract_kinematics::numericalJacobian(
305  numerical_jacobian, change_base, kin, jvals, kin.getTipLinkName(), link_point);
306  else
307  tesseract_kinematics::numericalJacobian(numerical_jacobian, change_base, kin, jvals, link_name, link_point);
308 
309  for (int i = 0; i < 6; ++i)
310  for (int j = 0; j < static_cast<int>(kin.numJoints()); ++j)
311  EXPECT_NEAR(numerical_jacobian(i, j), jacobian(i, j), 1e-3);
312 }
