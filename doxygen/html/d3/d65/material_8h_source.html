108 
109  std::string material_name;

111  return std::make_shared<SC>(MaterialStatusCategory::ErrorAttributeName, status_cat);
112 
113  auto m = std::make_shared<tesseract_scene_graph::Material>(material_name);
114

116  const tinyxml2::XMLElement* texture = xml_element->FirstChildElement("texture");
117  if (texture != nullptr)
118  {
119  if (tesseract_common::QueryStringAttribute(texture, "filename", m->texture_filename) != tinyxml2::XML_SUCCESS)
120  return std::make_shared<SC>(MaterialStatusCategory::ErrorTextureAttributeFilename, status_cat);
121  }
122 
123  const tinyxml2::XMLElement* color = xml_element->FirstChildElement("color");
124  if (color != nullptr)
125  {
126  std::string color_string;
127  if (tesseract_common::QueryStringAttribute(color, "rgba", color_string) != tinyxml2::XML_SUCCESS)
128  return std::make_shared<SC>(MaterialStatusCategory::ErrorColorAttributeRGBA, status_cat);
129 
130  if (!color_string.empty())
131  {
132  std::vector<std::string> tokens;
133  boost::split(tokens, color_string, boost::is_any_of(" "), boost::token_compress_on);
134  if (tokens.size() != 4 || !tesseract_common::isNumeric(tokens))
135  return std::make_shared<SC>(MaterialStatusCategory::ErrorParsingColorAttributeRGBA, status_cat);
136 
137  double r, g, b, a;
138  // No need to check return values because the tokens are verified above
139  tesseract_common::toNumeric<double>(tokens[0], r);
140  tesseract_common::toNumeric<double>(tokens[1], g);
141  tesseract_common::toNumeric<double>(tokens[2], b);
142  tesseract_common::toNumeric<double>(tokens[3], a);
143

145  }
146  else

149  }
150  }

157  }

178  }
179  }
