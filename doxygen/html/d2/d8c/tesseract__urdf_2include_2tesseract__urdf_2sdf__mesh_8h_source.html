103  if (tokens.size() != 3 || !tesseract_common::isNumeric(tokens))
104  return std::make_shared<tesseract_common::StatusCode>(SDFMeshStatusCategory::ErrorParsingAttributeScale,
105  status_cat);
106

108  // No need to check return values because the tokens are verified above
109  tesseract_common::toNumeric<double>(tokens[0], sx);
110  tesseract_common::toNumeric<double>(tokens[1], sy);
111  tesseract_common::toNumeric<double>(tokens[2], sz);
112

114  }
115

121  locator->locateResource(filename), scale, true, false);
