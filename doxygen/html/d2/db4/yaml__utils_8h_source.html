120  static bool decode(const Node& node, Eigen::Isometry3d& rhs)
121  {
122  Eigen::Isometry3d out = Eigen::Isometry3d::Identity();
123 
124  const YAML::Node& p = node["position"];
125  out.translation().x() = p["x"].as<double>();
126  out.translation().y() = p["y"].as<double>();
127  out.translation().z() = p["z"].as<double>();
128 
129  const YAML::Node& o = node["orientation"];
130  if (o["x"] && o["y"] && o["z"] && o["w"])
131  {
132  Eigen::Quaterniond quat;
133  quat.x() = o["x"].as<double>();
134  quat.y() = o["y"].as<double>();
135  quat.z() = o["z"].as<double>();
136  quat.w() = o["w"].as<double>();
137 
138  out.linear() = quat.toRotationMatrix();
139  }
140  else if (o["r"] && o["p"] && o["y"])
141  {
142  double r = o["r"].as<double>();
143  double p = o["p"].as<double>();
144  double y = o["y"].as<double>();
145 
146  Eigen::AngleAxisd rollAngle(r, Eigen::Vector3d::UnitX());
147  Eigen::AngleAxisd pitchAngle(p, Eigen::Vector3d::UnitY());
148  Eigen::AngleAxisd yawAngle(y, Eigen::Vector3d::UnitZ());
149 
150  Eigen::Quaterniond rpy = yawAngle * pitchAngle * rollAngle;
151 
152  out.linear() = rpy.toRotationMatrix();
153  }
154  else
155  {
156  throw std::runtime_error("Eigen::Isometry3d, failed to decode orientation missing (x, y, z, w) or (r, p, y)");
157  }
158 
159  rhs = out;
160  return true;
