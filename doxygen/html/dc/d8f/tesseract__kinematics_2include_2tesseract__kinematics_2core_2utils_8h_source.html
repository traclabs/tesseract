139 inline static bool solvePInv(const Eigen::Ref<const Eigen::MatrixXd>& A,
140  const Eigen::Ref<const Eigen::VectorXd>& b,
141  Eigen::Ref<Eigen::VectorXd> x)
142 {
143  const double eps = 0.00001; // TODO: Turn into class member var
144  const double lambda = 0.01; // TODO: Turn into class member var
145 
146  if ((A.rows() == 0) || (A.cols() == 0))
147  {
148  CONSOLE_BRIDGE_logError("Empty matrices not supported in solvePinv()");
149  return false;
150  }
151 
152  if (A.rows() != b.size())
153  {
154  CONSOLE_BRIDGE_logError("Matrix size mismatch: A(%d, %d), b(%d)", A.rows(), A.cols(), b.size());
155  return false;
156  }
157 
158  // Calculate A+ (pseudoinverse of A) = V S+ U*, where U* is Hermition of U (just transpose if all values of U are
159  // real)
160  // in order to solve Ax=b -> x*=A+ b
161  Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
162  const Eigen::MatrixXd& U = svd.matrixU();
163  const Eigen::VectorXd& Sv = svd.singularValues();
164  const Eigen::MatrixXd& V = svd.matrixV();
165 
166  // calculate the reciprocal of Singular-Values
167  // damp inverse with lambda so that inverse doesn't oscillate near solution
168  long int nSv = Sv.size();
169  Eigen::VectorXd inv_Sv(nSv);
170  for (long int i = 0; i < nSv; ++i)
171  {
172  if (fabs(Sv(i)) > eps)
173  inv_Sv(i) = 1 / Sv(i);
174  else
175  inv_Sv(i) = Sv(i) / (Sv(i) * Sv(i) + lambda * lambda);
176  }
177  x = V * inv_Sv.asDiagonal() * U.transpose() * b;
178  return true;
179 }

302 inline Manipulability calcManipulability(const Eigen::Ref<const Eigen::MatrixXd>& jacobian)
303 {
304  Manipulability manip;
305  Eigen::MatrixXd jacob_linear = jacobian.topRows(3);
306  Eigen::MatrixXd jacob_angular = jacobian.bottomRows(3);
307 
308  auto fn = [](const Eigen::MatrixXd& m) {
309  ManipulabilityEllipsoid data;
310  Eigen::EigenSolver<Eigen::MatrixXd> sm(m, false);
311 
312  data.eigen_values = sm.eigenvalues().real();
313  data.measure = std::sqrt(data.eigen_values.maxCoeff()) / std::sqrt(data.eigen_values.minCoeff());
314  data.condition = data.eigen_values.maxCoeff() / data.eigen_values.minCoeff();
315  data.volume = 1;
316  for (Eigen::Index i = 0; i < sm.eigenvalues().size(); ++i)
317  data.volume = data.volume * data.eigen_values[i];
318  data.volume = std::sqrt(data.volume);
319  return data;
320  };
321 
322  Eigen::MatrixXd a = jacobian * jacobian.transpose();
323  Eigen::MatrixXd a_linear = jacob_linear * jacob_linear.transpose();
324  Eigen::MatrixXd a_angular = jacob_angular * jacob_angular.transpose();
325  manip.m = fn(a);
326  manip.m_linear = fn(a_linear);
327  manip.m_angular = fn(a_angular);
328 
329  Eigen::MatrixXd a_inv = a.inverse();
330  Eigen::MatrixXd a_linear_inv = a_linear.inverse();
331  Eigen::MatrixXd a_angular_inv = a_angular.inverse();
332  manip.f = fn(a_inv);
333  manip.f_linear = fn(a_linear_inv);
334  manip.f_angular = fn(a_angular_inv);
335 
336  return manip;
337 }

404 template <typename FloatType>
405 inline void getRedundantSolutionsHelper(std::vector<VectorX<FloatType>>& redundant_sols,
406  const Eigen::Ref<const Eigen::VectorXd>& sol,
407  const Eigen::MatrixX2d& limits,
408  std::vector<Eigen::Index>::const_iterator current_index,
409  std::vector<Eigen::Index>::const_iterator end_index)
410 {
411  double val{ 0 };
412  for (; current_index != end_index; ++current_index)
413  {
414  if (std::isinf(limits(*current_index, 0)))
415  {
416  std::stringstream ss;
417  ss << "Lower limit of joint " << *current_index << " is infinite; no redundant solutions will be generated"
418  << std::endl;
419  CONSOLE_BRIDGE_logWarn(ss.str().c_str());
420  }
421  else
422  {
423  val = sol[*current_index];
424  while ((val -= (2.0 * M_PI)) > limits(*current_index, 0) ||
425  tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 0)))
426  {
427  // It not guaranteed that the provided solution is within limits so this check is needed
428  if (val < limits(*current_index, 1) ||
429  tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 1)))
430  {
431  Eigen::VectorXd new_sol = sol;
432  new_sol[*current_index] = val;
433 
434  if (tesseract_common::satisfiesPositionLimits(new_sol, limits))
435  {
436  tesseract_common::enforcePositionLimits(new_sol, limits);
437  redundant_sols.push_back(new_sol.template cast<FloatType>());
438  }
439 
440  getRedundantSolutionsHelper<FloatType>(redundant_sols, new_sol, limits, current_index + 1, end_index);
441  }
442  }
443  }
444 
445  if (std::isinf(limits(*current_index, 1)))
446  {
447  std::stringstream ss;
448  ss << "Upper limit of joint " << *current_index << " is infinite; no redundant solutions will be generated"
449  << std::endl;
450  CONSOLE_BRIDGE_logWarn(ss.str().c_str());
451  }
452  else
453  {
454  val = sol[*current_index];
455  while ((val += (2.0 * M_PI)) < limits(*current_index, 1) ||
456  tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 1)))
457  {
458  // It not guaranteed that the provided solution is within limits so this check is needed
459  if (val > limits(*current_index, 0) ||
460  tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 0)))
461  {
462  Eigen::VectorXd new_sol = sol;
463  new_sol[*current_index] = val;
464 
465  if (tesseract_common::satisfiesPositionLimits(new_sol, limits))
466  {
467  tesseract_common::enforcePositionLimits(new_sol, limits);
468  redundant_sols.push_back(new_sol.template cast<FloatType>());
469  }
470 
471  getRedundantSolutionsHelper<FloatType>(redundant_sols, new_sol, limits, current_index + 1, end_index);
472  }
473  }
474  }
475  }
476 }

486 inline std::vector<VectorX<FloatType>> getRedundantSolutions(const Eigen::Ref<const VectorX<FloatType>>& sol,
487  const Eigen::MatrixX2d& limits,
488  const std::vector<Eigen::Index>& redundancy_capable_joints)
489 {
490  if (redundancy_capable_joints.empty())
491  return {};
492 
493  for (const Eigen::Index& idx : redundancy_capable_joints)
494  {
495  if (idx >= sol.size())
496  {
497  std::stringstream ss;
498  ss << "Redundant joint index " << idx << " is greater than or equal to the joint state size (" << sol.size()
499  << ")";
500  throw std::runtime_error(ss.str());
501  }
502  }
503 
504  std::vector<VectorX<FloatType>> redundant_sols;
505  getRedundantSolutionsHelper<FloatType>(redundant_sols,
506  sol.template cast<double>(),
507  limits,
508  redundancy_capable_joints.begin(),
509  redundancy_capable_joints.end());
510  return redundant_sols;
511 }
