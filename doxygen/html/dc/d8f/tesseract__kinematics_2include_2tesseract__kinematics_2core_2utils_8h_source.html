83 inline static void numericalJacobian(Eigen::Ref<Eigen::MatrixXd> jacobian,
84  const Eigen::Isometry3d& change_base,
85  const tesseract_kinematics::ForwardKinematics& kin,
86  const Eigen::Ref<const Eigen::VectorXd>& joint_values,
87  const std::string& link_name,
88  const Eigen::Ref<const Eigen::Vector3d>& link_point)
89 {
90  Eigen::VectorXd njvals;
91  double delta = 0.001;
92  Eigen::Isometry3d pose;
93  kin.calcFwdKin(pose, joint_values, link_name);
94  pose = change_base * pose;
95 
96  for (int i = 0; i < static_cast<int>(joint_values.size()); ++i)
97  {
98  njvals = joint_values;
99  njvals[i] += delta;
100  Eigen::Isometry3d updated_pose;
101  kin.calcFwdKin(updated_pose, njvals, link_name);
102  updated_pose = change_base * updated_pose;
103 
104  Eigen::Vector3d temp = pose * link_point;
105  Eigen::Vector3d temp2 = updated_pose * link_point;
106  jacobian(0, i) = (temp2.x() - temp.x()) / delta;
107  jacobian(1, i) = (temp2.y() - temp.y()) / delta;
108  jacobian(2, i) = (temp2.z() - temp.z()) / delta;
109 
110  Eigen::AngleAxisd r12(pose.rotation().transpose() * updated_pose.rotation()); // rotation from p1 -> p2
111  double theta = r12.angle();
112  theta = copysign(fmod(fabs(theta), 2.0 * M_PI), theta);
113  if (theta < -M_PI)
114  theta = theta + 2. * M_PI;
115  if (theta > M_PI)
116  theta = theta - 2. * M_PI;
117  Eigen::VectorXd omega = (pose.rotation() * r12.axis() * theta) / delta;
118  jacobian(3, i) = omega(0);
119  jacobian(4, i) = omega(1);
120  jacobian(5, i) = omega(2);
121  }
122 }

132 inline static bool solvePInv(const Eigen::Ref<const Eigen::MatrixXd>& A,
133  const Eigen::Ref<const Eigen::VectorXd>& b,
134  Eigen::Ref<Eigen::VectorXd> x)
135 {
136  const double eps = 0.00001; // TODO: Turn into class member var
137  const double lambda = 0.01; // TODO: Turn into class member var
138 
139  if ((A.rows() == 0) || (A.cols() == 0))
140  {
141  CONSOLE_BRIDGE_logError("Empty matrices not supported in solvePinv()");
142  return false;
143  }
144 
145  if (A.rows() != b.size())
146  {
147  CONSOLE_BRIDGE_logError("Matrix size mismatch: A(%d, %d), b(%d)", A.rows(), A.cols(), b.size());
148  return false;
149  }
150 
151  // Calculate A+ (pseudoinverse of A) = V S+ U*, where U* is Hermition of U (just transpose if all values of U are
152  // real)
153  // in order to solve Ax=b -> x*=A+ b
154  Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
155  const Eigen::MatrixXd& U = svd.matrixU();
156  const Eigen::VectorXd& Sv = svd.singularValues();
157  const Eigen::MatrixXd& V = svd.matrixV();
158 
159  // calculate the reciprocal of Singular-Values
160  // damp inverse with lambda so that inverse doesn't oscillate near solution
161  long int nSv = Sv.size();
162  Eigen::VectorXd inv_Sv(nSv);
163  for (long int i = 0; i < nSv; ++i)
164  {
165  if (fabs(Sv(i)) > eps)
166  inv_Sv(i) = 1 / Sv(i);
167  else
168  inv_Sv(i) = Sv(i) / (Sv(i) * Sv(i) + lambda * lambda);
169  }
170  x = V * inv_Sv.asDiagonal() * U.transpose() * b;
171  return true;
172 }

184  Eigen::Ref<Eigen::MatrixXd> P,

305 {

307  FloatType val;
308  std::vector<FloatType> redundant_sols;

313  {
314  std::vector<FloatType> new_sol(sol, sol + dof);
315  new_sol[static_cast<size_t>(i)] = val;
316  redundant_sols.insert(redundant_sols.end(), new_sol.begin(), new_sol.end());
317  }

323  new_sol[static_cast<size_t>(i)] = val;
