88 inline static void numericalJacobian(Eigen::Ref<Eigen::MatrixXd> jacobian,
89  const Eigen::Isometry3d& change_base,
90  const tesseract_kinematics::ForwardKinematics& kin,
91  const Eigen::Ref<const Eigen::VectorXd>& joint_values,
92  const std::string& link_name,
93  const Eigen::Ref<const Eigen::Vector3d>& link_point)
94 {
95  Eigen::VectorXd njvals;
96  double delta = 0.001;
97  Eigen::Isometry3d pose = kin.calcFwdKin(joint_values, link_name);
98  pose = change_base * pose;
99 
100  for (int i = 0; i < static_cast<int>(joint_values.size()); ++i)
101  {
102  njvals = joint_values;
103  njvals[i] += delta;
104  Eigen::Isometry3d updated_pose = kin.calcFwdKin(njvals, link_name);
105  updated_pose = change_base * updated_pose;
106 
107  Eigen::Vector3d temp = pose * link_point;
108  Eigen::Vector3d temp2 = updated_pose * link_point;
109  jacobian(0, i) = (temp2.x() - temp.x()) / delta;
110  jacobian(1, i) = (temp2.y() - temp.y()) / delta;
111  jacobian(2, i) = (temp2.z() - temp.z()) / delta;
112 
113  Eigen::AngleAxisd r12(pose.rotation().transpose() * updated_pose.rotation()); // rotation from p1 -> p2
114  double theta = r12.angle();
115  theta = copysign(fmod(fabs(theta), 2.0 * M_PI), theta);
116  if (theta < -M_PI)
117  theta = theta + 2. * M_PI;
118  if (theta > M_PI)
119  theta = theta - 2. * M_PI;
120  Eigen::VectorXd omega = (pose.rotation() * r12.axis() * theta) / delta;
121  jacobian(3, i) = omega(0);
122  jacobian(4, i) = omega(1);
123  jacobian(5, i) = omega(2);
124  }

135 inline static bool solvePInv(const Eigen::Ref<const Eigen::MatrixXd>& A,
136  const Eigen::Ref<const Eigen::VectorXd>& b,
137  Eigen::Ref<Eigen::VectorXd> x)
138 {
139  const double eps = 0.00001; // TODO: Turn into class member var
140  const double lambda = 0.01; // TODO: Turn into class member var
141 
142  if ((A.rows() == 0) || (A.cols() == 0))
143  {
144  CONSOLE_BRIDGE_logError("Empty matrices not supported in solvePinv()");
145  return false;
146  }
147 
148  if (A.rows() != b.size())
149  {
150  CONSOLE_BRIDGE_logError("Matrix size mismatch: A(%d, %d), b(%d)", A.rows(), A.cols(), b.size());
151  return false;
152  }
153 
154  // Calculate A+ (pseudoinverse of A) = V S+ U*, where U* is Hermition of U (just transpose if all values of U are
155  // real)
156  // in order to solve Ax=b -> x*=A+ b
157  Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
158  const Eigen::MatrixXd& U = svd.matrixU();
159  const Eigen::VectorXd& Sv = svd.singularValues();
160  const Eigen::MatrixXd& V = svd.matrixV();
161 
162  // calculate the reciprocal of Singular-Values
163  // damp inverse with lambda so that inverse doesn't oscillate near solution
164  long int nSv = Sv.size();
165  Eigen::VectorXd inv_Sv(nSv);
166  for (long int i = 0; i < nSv; ++i)
167  {
168  if (fabs(Sv(i)) > eps)
169  inv_Sv(i) = 1 / Sv(i);
170  else
171  inv_Sv(i) = Sv(i) / (Sv(i) * Sv(i) + lambda * lambda);
172  }
173  x = V * inv_Sv.asDiagonal() * U.transpose() * b;
174  return true;
175 }

298 inline Manipulability calcManipulability(const Eigen::Ref<const Eigen::MatrixXd>& jacobian)
299 {
300  Manipulability manip;
301  Eigen::MatrixXd jacob_linear = jacobian.topRows(3);
302  Eigen::MatrixXd jacob_angular = jacobian.bottomRows(3);
303 
304  auto fn = [](const Eigen::MatrixXd& m) {
305  ManipulabilityEllipsoid data;
306  Eigen::EigenSolver<Eigen::MatrixXd> sm(m, false);
307 
308  data.eigen_values = sm.eigenvalues().real();
309  data.measure = std::sqrt(data.eigen_values.maxCoeff()) / std::sqrt(data.eigen_values.minCoeff());
310  data.condition = data.eigen_values.maxCoeff() / data.eigen_values.minCoeff();
311  data.volume = 1;
312  for (Eigen::Index i = 0; i < sm.eigenvalues().size(); ++i)
313  data.volume = data.volume * data.eigen_values[i];
314  data.volume = std::sqrt(data.volume);
315  return data;
316  };
317 
318  Eigen::MatrixXd a = jacobian * jacobian.transpose();
319  Eigen::MatrixXd a_linear = jacob_linear * jacob_linear.transpose();
320  Eigen::MatrixXd a_angular = jacob_angular * jacob_angular.transpose();
321  manip.m = fn(a);
322  manip.m_linear = fn(a_linear);
323  manip.m_angular = fn(a_angular);
324 
325  Eigen::MatrixXd a_inv = a.inverse();
326  Eigen::MatrixXd a_linear_inv = a_linear.inverse();
327  Eigen::MatrixXd a_angular_inv = a_angular.inverse();
328  manip.f = fn(a_inv);
329  manip.f_linear = fn(a_linear_inv);
330  manip.f_angular = fn(a_angular_inv);
331 
332  return manip;
333 }

401 inline void getRedundantSolutionsHelper(std::vector<VectorX<FloatType>>& redundant_sols,
402  const Eigen::Ref<const Eigen::VectorXd>& sol,
403  const Eigen::MatrixX2d& limits,
404  std::vector<Eigen::Index>::const_iterator current_index,
405  std::vector<Eigen::Index>::const_iterator end_index)
406 {
407  double val;
408  for (; current_index != end_index; ++current_index)
409  {
410  if (std::isinf(limits(*current_index, 0)))
411  {
412  std::stringstream ss;
413  ss << "Lower limit of joint " << *current_index << " is infinite; no redundant solutions will be generated"
414  << std::endl;
415  CONSOLE_BRIDGE_logWarn(ss.str().c_str());
416  }
417  else
418  {
419  val = sol[*current_index];
420  while ((val -= (2.0 * M_PI)) > limits(*current_index, 0) ||
421  tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 0)))
422  {
423  // It not guaranteed that the provided solution is within limits so this check is needed
424  if (val < limits(*current_index, 1) ||
425  tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 1)))
426  {
427  Eigen::VectorXd new_sol = sol;
428  new_sol[*current_index] = val;
429 
430  if (tesseract_common::satisfiesPositionLimits(new_sol, limits))
431  {
432  tesseract_common::enforcePositionLimits(new_sol, limits);
433  redundant_sols.push_back(new_sol.template cast<FloatType>());
434  }
435 
436  getRedundantSolutionsHelper<FloatType>(redundant_sols, new_sol, limits, current_index + 1, end_index);
437  }
438  }
439  }
440 
441  if (std::isinf(limits(*current_index, 1)))
442  {
443  std::stringstream ss;
444  ss << "Upper limit of joint " << *current_index << " is infinite; no redundant solutions will be generated"
445  << std::endl;
446  CONSOLE_BRIDGE_logWarn(ss.str().c_str());
447  }
448  else
449  {
450  val = sol[*current_index];
451  while ((val += (2.0 * M_PI)) < limits(*current_index, 1) ||
452  tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 1)))
453  {
454  // It not guaranteed that the provided solution is within limits so this check is needed
455  if (val > limits(*current_index, 0) ||

457  {
458  Eigen::VectorXd new_sol = sol;
459  new_sol[*current_index] = val;
460

464  redundant_sols.push_back(new_sol.template cast<FloatType>());
465  }
