169 inline int createConvexHull(tesseract_common::VectorVector3d& vertices,
170  Eigen::VectorXi& faces,
171  const tesseract_common::VectorVector3d& input,
172  double shrink = -1,
173  double shrinkClamp = -1)
174 {
175  vertices.clear();
176 
177  btConvexHullComputer conv;
178  std::vector<double> points;
179  points.reserve(input.size() * 3);
180  for (const auto& v : input)
181  {
182  points.push_back(v[0]);
183  points.push_back(v[1]);
184  points.push_back(v[2]);
185  }
186 
187  btScalar val = conv.compute(points.data(),
188  3 * sizeof(double),
189  static_cast<int>(input.size()),
190  static_cast<btScalar>(shrink),
191  static_cast<btScalar>(shrinkClamp));
192  if (val < 0)
193  {
194  CONSOLE_BRIDGE_logError("Failed to create convex hull");
195  return -1;
196  }
197 
198  int num_verts = conv.vertices.size();
199  vertices.reserve(static_cast<size_t>(num_verts));
200  for (int i = 0; i < num_verts; i++)
201  {
202  btVector3& v = conv.vertices[i];
203  vertices.push_back(
204  Eigen::Vector3d(static_cast<double>(v.getX()), static_cast<double>(v.getY()), static_cast<double>(v.getZ())));
205  }
206 
207  auto num_faces = static_cast<size_t>(conv.faces.size());
208  std::vector<int> local_faces;
209  local_faces.reserve(3ul * num_faces);
210  for (int i = 0; i < conv.faces.size(); i++)
211  {
212  std::vector<int> face;
213  face.reserve(3);
214 
215  const btConvexHullComputer::Edge* sourceEdge = &(conv.edges[conv.faces[i]]);
216  int a = sourceEdge->getSourceVertex();
217  face.push_back(a);
218 
219  int b = sourceEdge->getTargetVertex();
220  face.push_back(b);
221 
222  const btConvexHullComputer::Edge* edge = sourceEdge->getNextEdgeOfFace();
223  int c = edge->getTargetVertex();
224  face.push_back(c);
225 
226  edge = edge->getNextEdgeOfFace();
227  c = edge->getTargetVertex();
228  while (c != a)
229  {
230  face.push_back(c);
231 
232  edge = edge->getNextEdgeOfFace();
233  c = edge->getTargetVertex();
234  }
235  local_faces.push_back(static_cast<int>(face.size()));
236  local_faces.insert(local_faces.end(), face.begin(), face.end());
237  }
238 
239  faces.resize(static_cast<long>(local_faces.size()));
240  for (size_t i = 0; i < local_faces.size(); ++i)
241  faces[static_cast<long>(i)] = local_faces[i];
242 
243  return conv.faces.size();
244 }
