224 inline int createConvexHull(tesseract_common::VectorVector3d& vertices,
225  Eigen::VectorXi& faces,
226  const tesseract_common::VectorVector3d& input,
227  double shrink = -1,
228  double shrinkClamp = -1)
229 {
230  vertices.clear();
231 
232  btConvexHullComputer conv;
233  std::vector<double> points;
234  points.reserve(input.size() * 3);
235  for (const auto& v : input)
236  {
237  points.push_back(v[0]);
238  points.push_back(v[1]);
239  points.push_back(v[2]);
240  }
241 
242  btScalar val = conv.compute(points.data(),
243  3 * sizeof(double),
244  static_cast<int>(input.size()),
245  static_cast<btScalar>(shrink),
246  static_cast<btScalar>(shrinkClamp));
247  if (val < 0)
248  {
249  CONSOLE_BRIDGE_logError("Failed to create convex hull");
250  return -1;
251  }
252 
253  int num_verts = conv.vertices.size();
254  vertices.reserve(static_cast<size_t>(num_verts));
255  for (int i = 0; i < num_verts; i++)
256  {
257  btVector3& v = conv.vertices[i];
258  vertices.push_back(
259  Eigen::Vector3d(static_cast<double>(v.getX()), static_cast<double>(v.getY()), static_cast<double>(v.getZ())));
260  }
261 
262  auto num_faces = static_cast<size_t>(conv.faces.size());
263  std::vector<int> local_faces;
264  local_faces.reserve(3UL * num_faces);
265  for (int i = 0; i < conv.faces.size(); i++)
266  {
267  std::vector<int> face;
268  face.reserve(3);
269 
270  const btConvexHullComputer::Edge* sourceEdge = &(conv.edges[conv.faces[i]]);
271  int a = sourceEdge->getSourceVertex();
272  face.push_back(a);
273 
274  int b = sourceEdge->getTargetVertex();
275  face.push_back(b);
276 
277  const btConvexHullComputer::Edge* edge = sourceEdge->getNextEdgeOfFace();
278  int c = edge->getTargetVertex();
279  face.push_back(c);
280 
281  edge = edge->getNextEdgeOfFace();
282  c = edge->getTargetVertex();
283  while (c != a)
284  {
285  face.push_back(c);
286 
287  edge = edge->getNextEdgeOfFace();
288  c = edge->getTargetVertex();
289  }
290  local_faces.push_back(static_cast<int>(face.size()));
291  local_faces.insert(local_faces.end(), face.begin(), face.end());
292  }
293 
294  faces.resize(static_cast<long>(local_faces.size()));
295  for (size_t i = 0; i < local_faces.size(); ++i)
296  faces[static_cast<long>(i)] = local_faces[i];
297 
298  return conv.faces.size();
299 }
