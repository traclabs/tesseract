181 inline int createConvexHull(tesseract_common::VectorVector3d& vertices,
182  Eigen::VectorXi& faces,
183  const tesseract_common::VectorVector3d& input,
184  double shrink = -1,
185  double shrinkClamp = -1)
186 {
187  vertices.clear();
188 
189  btConvexHullComputer conv;
190  std::vector<double> points;
191  points.reserve(input.size() * 3);
192  for (const auto& v : input)
193  {
194  points.push_back(v[0]);
195  points.push_back(v[1]);
196  points.push_back(v[2]);
197  }
198 
199  btScalar val = conv.compute(points.data(),
200  3 * sizeof(double),
201  static_cast<int>(input.size()),
202  static_cast<btScalar>(shrink),
203  static_cast<btScalar>(shrinkClamp));
204  if (val < 0)
205  {
206  CONSOLE_BRIDGE_logError("Failed to create convex hull");
207  return -1;
208  }
209 
210  int num_verts = conv.vertices.size();
211  vertices.reserve(static_cast<size_t>(num_verts));
212  for (int i = 0; i < num_verts; i++)
213  {
214  btVector3& v = conv.vertices[i];
215  vertices.push_back(
216  Eigen::Vector3d(static_cast<double>(v.getX()), static_cast<double>(v.getY()), static_cast<double>(v.getZ())));
217  }
218 
219  auto num_faces = static_cast<size_t>(conv.faces.size());
220  std::vector<int> local_faces;
221  local_faces.reserve(3ul * num_faces);
222  for (int i = 0; i < conv.faces.size(); i++)
223  {
224  std::vector<int> face;
225  face.reserve(3);
226 
227  const btConvexHullComputer::Edge* sourceEdge = &(conv.edges[conv.faces[i]]);
228  int a = sourceEdge->getSourceVertex();
229  face.push_back(a);
230 
231  int b = sourceEdge->getTargetVertex();
232  face.push_back(b);
233 
234  const btConvexHullComputer::Edge* edge = sourceEdge->getNextEdgeOfFace();
235  int c = edge->getTargetVertex();
236  face.push_back(c);
237 
238  edge = edge->getNextEdgeOfFace();
239  c = edge->getTargetVertex();
240  while (c != a)
241  {
242  face.push_back(c);
243 
244  edge = edge->getNextEdgeOfFace();
245  c = edge->getTargetVertex();
246  }
247  local_faces.push_back(static_cast<int>(face.size()));
248  local_faces.insert(local_faces.end(), face.begin(), face.end());
249  }
250 
251  faces.resize(static_cast<long>(local_faces.size()));
252  for (size_t i = 0; i < local_faces.size(); ++i)
253  faces[static_cast<long>(i)] = local_faces[i];
254 
255  return conv.faces.size();
256 }
