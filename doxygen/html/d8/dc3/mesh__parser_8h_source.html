75 template <class T>
76 std::vector<std::shared_ptr<T>> extractMeshData(const aiScene* scene,
77  const aiNode* node,
78  const aiMatrix4x4& parent_transform,
79  const Eigen::Vector3d& scale,
80  tesseract_common::Resource::Ptr resource,
81  bool normals,
82  bool vertex_colors,
83  bool material_and_texture)
84 {
85  std::vector<std::shared_ptr<T>> meshes;
86 
87  aiMatrix4x4 transform = parent_transform;
88  transform *= node->mTransformation;
89  for (unsigned int j = 0; j < node->mNumMeshes; ++j)
90  {
91  auto vertices = std::make_shared<tesseract_common::VectorVector3d>();
92  auto triangles = std::make_shared<Eigen::VectorXi>();
93  std::shared_ptr<tesseract_common::VectorVector3d> vertex_normals = nullptr;
94  std::shared_ptr<tesseract_common::VectorVector4d> vertex_colors = nullptr;
95  MeshMaterial::Ptr material = nullptr;
96  std::shared_ptr<std::vector<MeshTexture::Ptr>> textures = nullptr;
97 
98  const aiMesh* a = scene->mMeshes[node->mMeshes[j]];
99  for (unsigned int i = 0; i < a->mNumVertices; ++i)
100  {
101  aiVector3D v = transform * a->mVertices[i];
102  vertices->push_back(Eigen::Vector3d(static_cast<double>(v.x) * scale(0),
103  static_cast<double>(v.y) * scale(1),
104  static_cast<double>(v.z) * scale(2)));
105  }
106 
107  long triangle_count = 0;
108  std::vector<int> local_triangles;
109  local_triangles.reserve(a->mNumFaces);
110  for (unsigned int i = 0; i < a->mNumFaces; ++i)
111  {
112  if (a->mFaces[i].mNumIndices >= 3)
113  {
114  triangle_count += 1;
115  local_triangles.push_back(static_cast<int>(a->mFaces[i].mNumIndices));
116  for (size_t k = 0; k < a->mFaces[i].mNumIndices; ++k)
117  local_triangles.push_back(static_cast<int>(a->mFaces[i].mIndices[k]));
118  }
119  else
120  {
121  CONSOLE_BRIDGE_logDebug("Mesh had a face with less than three verticies: %s", resource->getUrl().c_str());
122  }
123  }
124 
125  triangles->resize(static_cast<long>(local_triangles.size()));
126  for (long i = 0; i < triangles->size(); ++i)
127  (*triangles)[i] = local_triangles[static_cast<size_t>(i)];
128 
129  if (normals && a->HasNormals())
130  {
131  vertex_normals = std::make_shared<tesseract_common::VectorVector3d>();
132  for (unsigned int i = 0; i < a->mNumVertices; ++i)
133  {
134  aiVector3D v = transform * a->mNormals[i];
135  vertex_normals->push_back(Eigen::Vector3d(static_cast<double>(v.x) * scale(0),
136  static_cast<double>(v.y) * scale(1),
137  static_cast<double>(v.z) * scale(2)));
138  }
139  }
140 
141  if (vertex_colors && a->HasVertexColors(0))
142  {
143  vertex_colors = std::make_shared<tesseract_common::VectorVector4d>();
144  for (unsigned int i = 0; i < a->mNumVertices; ++i)
145  {
146  aiColor4D v = a->mColors[0][i];
147  vertex_colors->push_back(Eigen::Vector4d(
148  static_cast<double>(v.r), static_cast<double>(v.g), static_cast<double>(v.b), static_cast<double>(v.a)));
149  }
150  }
151 
152  if (material_and_texture)
153  {
154  aiMaterial* mat = scene->mMaterials[a->mMaterialIndex];
155  {
156  Eigen::Vector4d base_color;
157  double metallic = 0.0;
158  double roughness = 0.5;
159  Eigen::Vector4d emissive;
160 
161  aiColor4D pbr_base_color;
162 #ifdef TESSERACT_ASSIMP_USE_PBRMATERIAL
163  if (mat->Get(AI_MATKEY_GLTF_PBRMETALLICROUGHNESS_BASE_COLOR_FACTOR, pbr_base_color) == AI_SUCCESS)
164  {
165  // Use PBR Metallic material properties if available
166  base_color = Eigen::Vector4d(pbr_base_color.r, pbr_base_color.g, pbr_base_color.b, pbr_base_color.a);
167  float metallicFactor{ 0 };
168  if (mat->Get(AI_MATKEY_GLTF_PBRMETALLICROUGHNESS_METALLIC_FACTOR, metallicFactor) == AI_SUCCESS)
169  {
170  metallic = metallicFactor;
171  }
172  float roughnessFactor{ 0.5 };
173  if (mat->Get(AI_MATKEY_GLTF_PBRMETALLICROUGHNESS_ROUGHNESS_FACTOR, roughnessFactor) == AI_SUCCESS)
174  {
175  roughness = roughnessFactor;
176  }
177  aiColor4D pbr_emissive_color;
178  if (mat->Get(AI_MATKEY_COLOR_EMISSIVE, pbr_emissive_color) == AI_SUCCESS)
179  {
180  emissive =
181  Eigen::Vector4d(pbr_emissive_color.r, pbr_emissive_color.g, pbr_emissive_color.b, pbr_emissive_color.a);
182  }
183  }
184  else
185 #endif
186  {
187  // Use old style material. Ambient and specular not supported
188  aiColor4D diffuse_color;
189  if (mat->Get(AI_MATKEY_COLOR_DIFFUSE, diffuse_color) == AI_SUCCESS)
190  {
191  base_color = Eigen::Vector4d(diffuse_color.r, diffuse_color.g, diffuse_color.b, diffuse_color.a);
192  }
193 
194  aiColor4D emissive_color;
195  if (mat->Get(AI_MATKEY_COLOR_EMISSIVE, emissive_color) == AI_SUCCESS)
196  {
197  emissive = Eigen::Vector4d(emissive_color.r, emissive_color.g, emissive_color.b, emissive_color.a);
198  }
199  }
200 
201  material = std::make_shared<MeshMaterial>(base_color, metallic, roughness, emissive);
202 
203  for (unsigned int i = 0; i < a->GetNumUVChannels(); i++)
204  {
205  if (a->HasTextureCoords(i))
206  {
207  aiString texName;
208  aiTextureMapping mapping{ aiTextureMapping_OTHER };
209  unsigned int uvIndex{ 0 };
210  if (mat->GetTexture(aiTextureType_DIFFUSE, i, &texName, &mapping, &uvIndex) == AI_SUCCESS)
211  {
212  tesseract_common::Resource::Ptr texture_image;
213  tesseract_common::VectorVector2d uvs;
214  // https://stackoverflow.com/questions/56820244/assimp-doenst-return-texture-data
215  const char* texNamec = texName.C_Str();
216  if ('*' == *texNamec)
217  {
218  int tex_index = std::atoi(texNamec + 1);
219  if (0 > tex_index || scene->mNumTextures <= static_cast<unsigned>(tex_index))
220  continue;
221  auto* texture_data = scene->mTextures[tex_index];
222  // returned pointer is not null, read texture from memory
223  std::string file_type = texture_data->achFormatHint;
224  if (file_type == "jpg" || file_type == "png")
225  {
226  texture_image = std::make_shared<tesseract_common::BytesResource>(
227  "data://", (const uint8_t*)texture_data->pcData, texture_data->mWidth); // NOLINT
228  }
229  else
230  {
231  // TODO: handle other file types
232  continue;
233  }
234  }
235  else
236  {
237  if (!resource)
238  {
239  continue;
240  }
241  std::string texName_str(texName.C_Str());
242  auto tex_resource = resource->locateResource(texName_str);
243  if (!tex_resource)
244  {
245  continue;
246  }
247  texture_image = tex_resource;
248  }
249  aiVector3D* tex_coords = a->mTextureCoords[i];
250  for (unsigned int j = 0; j < a->mNumVertices; ++j)
251  {
252  aiVector3D v = tex_coords[j];
253  uvs.push_back(Eigen::Vector2d(static_cast<double>(v.x), static_cast<double>(v.y)));
254  }
255  auto tex = std::make_shared<MeshTexture>(
256  texture_image, std::make_shared<tesseract_common::VectorVector2d>(std::move(uvs)));
257  if (!textures)
258  {
259  textures = std::make_shared<std::vector<MeshTexture::Ptr>>();
260  }
261  textures->push_back(tex);
262  }
263  }
264  }
265  }
266  }
267 
268  meshes.push_back(std::make_shared<T>(vertices,
269  triangles,
270  static_cast<int>(triangle_count),
271  resource,
272  scale,
273  vertex_normals,
274  vertex_colors,
275  material,
276  textures));
277  }
278 
279  for (unsigned int n = 0; n < node->mNumChildren; ++n)
280  {
281  std::vector<std::shared_ptr<T>> child_meshes = extractMeshData<T>(
282  scene, node->mChildren[n], transform, scale, resource, normals, vertex_colors, material_and_texture);
283  meshes.insert(meshes.end(), child_meshes.begin(), child_meshes.end());
284  }
285  return meshes;
286 }

298 template <class T>
299 std::vector<std::shared_ptr<T>> createMeshFromAsset(const aiScene* scene,
300  const Eigen::Vector3d& scale,
301  tesseract_common::Resource::Ptr resource,
302  bool normals,
303  bool vertex_colors,
304  bool material_and_texture)
305 {
306  if (!scene->HasMeshes())
307  {
308  CONSOLE_BRIDGE_logWarn("Assimp reports scene in %s has no meshes", resource->getUrl().c_str());
309  return std::vector<std::shared_ptr<T>>();
310  }
311  std::vector<std::shared_ptr<T>> meshes = extractMeshData<T>(
312  scene, scene->mRootNode, aiMatrix4x4(), scale, resource, normals, vertex_colors, material_and_texture);
313  if (meshes.empty())
314  {
315  CONSOLE_BRIDGE_logWarn("There are no meshes in the scene %s", resource->getUrl().c_str());
316  return std::vector<std::shared_ptr<T>>();
317  }
318 
319  return meshes;
320 }

335 template <class T>
336 std::vector<std::shared_ptr<T>> createMeshFromPath(const std::string& path,
337  Eigen::Vector3d scale = Eigen::Vector3d(1, 1, 1),
338  bool triangulate = false,
339  bool flatten = false,
340  bool normals = false,
341  bool vertex_colors = false,
342  bool material_and_texture = false)
343 {
344  // Create an instance of the Importer class
345  Assimp::Importer importer;
346 
347  // Issue #38 fix: as part of the post-processing, we remove all other components in file but
348  // the meshes, as anyway the resulting shapes:Mesh object just receives vertices and triangles.
349  // John Wason Jan 2021 - Adjust flags based on normals, vertex_colors, and material_and_texture parameters
350  unsigned int ai_config_pp_rcv_flags = aiComponent_TANGENTS_AND_BITANGENTS | aiComponent_BONEWEIGHTS |
351  aiComponent_ANIMATIONS | aiComponent_LIGHTS | aiComponent_CAMERAS;
352  if (!normals)
353  {
354  ai_config_pp_rcv_flags |= aiComponent_NORMALS;
355  }
356  if (!vertex_colors)
357  {
358  ai_config_pp_rcv_flags |= aiComponent_COLORS;
359  }
360  if (!material_and_texture)
361  {
362  ai_config_pp_rcv_flags |= aiComponent_TEXCOORDS | aiComponent_TEXTURES | aiComponent_MATERIALS;
363  }
364  importer.SetPropertyInteger(AI_CONFIG_PP_RVC_FLAGS, (int)ai_config_pp_rcv_flags);
365 
366  // And have it read the given file with some post-processing
367  const aiScene* scene = nullptr;
368  if (triangulate)
369  scene = importer.ReadFile(path.c_str(),
370  aiProcess_Triangulate | aiProcess_JoinIdenticalVertices | aiProcess_SortByPType |
371  aiProcess_RemoveComponent);
372  else
373  scene = importer.ReadFile(path.c_str(),
374  aiProcess_JoinIdenticalVertices | aiProcess_SortByPType | aiProcess_RemoveComponent);
375 
376  if (!scene)
377  {
378  CONSOLE_BRIDGE_logError("Could not load mesh from \"%s\": %s", path.c_str(), importer.GetErrorString());
379  return std::vector<std::shared_ptr<T>>();
380  }
381 
382  // Assimp enforces Y_UP convention by rotating models with different conventions.
383  // However, that behaviour is confusing and doesn't match the ROS convention
384  // where the Z axis is pointing up.
385  // Hopefully this doesn't undo legit use of the root node transformation...
386  // Note that this is also what RViz does internally.
387  // @todo See this issue for possible fix parsing scene metadata https://github.com/assimp/assimp/issues/849
388  scene->mRootNode->mTransformation = aiMatrix4x4();
389 
390  if (flatten)
391  {
392  // These post processing steps flatten the root node transformation into child nodes,
393  // so they must be delayed until after clearing the root node transform above.
394  importer.ApplyPostProcessing(aiProcess_OptimizeMeshes | aiProcess_OptimizeGraph);
395  }
396  else
397  {
398  importer.ApplyPostProcessing(aiProcess_OptimizeGraph);
399  }
400 
401  return createMeshFromAsset<T>(scene, scale, nullptr, normals, vertex_colors, material_and_texture);
402 }

417 template <class T>
418 std::vector<std::shared_ptr<T>> createMeshFromResource(tesseract_common::Resource::Ptr resource,
419  Eigen::Vector3d scale = Eigen::Vector3d(1, 1, 1),
420  bool triangulate = false,
421  bool flatten = false,
422  bool normals = false,
423  bool vertex_colors = false,
424  bool material_and_texture = false)
425 {
426  if (!resource)
427  return std::vector<std::shared_ptr<T>>();
428 
429  const char* hint = nullptr;
430  std::string hint_storage;
431 
432  std::string resource_url = resource->getUrl();
433  std::regex hint_re("^.*\\.([A-Za-z0-9]{1,8})$");
434  std::smatch hint_match;
435  if (std::regex_match(resource_url, hint_match, hint_re))
436  {
437  if (hint_match.size() == 2)
438  {
439  hint_storage = hint_match[1].str();
440  hint = hint_storage.c_str();
441  }
442  }
443 
444  std::vector<uint8_t> data = resource->getResourceContents();
445  if (data.empty())
446  {
447  if (resource->isFile())
448  return createMeshFromPath<T>(
449  resource->getFilePath(), scale, triangulate, flatten, normals, vertex_colors, material_and_texture);
450 
451  return std::vector<std::shared_ptr<T>>();
452  }
453

455  Assimp::Importer importer;
456 
457  // Issue #38 fix: as part of the post-processing, we remove all other components in file but
458  // the meshes, as anyway the resulting shapes:Mesh object just receives vertices and triangles.
459  // John Wason Jan 2021 - Adjust flags based on normals, vertex_colors, and material_and_texture parameters
460  unsigned int ai_config_pp_rcv_flags = aiComponent_TANGENTS_AND_BITANGENTS | aiComponent_BONEWEIGHTS |
461  aiComponent_ANIMATIONS | aiComponent_LIGHTS | aiComponent_CAMERAS;
462  if (!normals)
463  {
464  ai_config_pp_rcv_flags |= aiComponent_NORMALS;
465  }
466  if (!vertex_colors)
467  {
468  ai_config_pp_rcv_flags |= aiComponent_COLORS;
469  }
470  if (!material_and_texture)
471  {
472  ai_config_pp_rcv_flags |= aiComponent_TEXCOORDS | aiComponent_TEXTURES | aiComponent_MATERIALS;
473  }
474  importer.SetPropertyInteger(AI_CONFIG_PP_RVC_FLAGS, (int)ai_config_pp_rcv_flags);
475 
476  // And have it read the given file with some post-processing
477  const aiScene* scene = nullptr;
478  if (triangulate)
479  scene = importer.ReadFileFromMemory(&data[0],
480  data.size(),
481  aiProcess_Triangulate | aiProcess_JoinIdenticalVertices |
482  aiProcess_SortByPType | aiProcess_RemoveComponent,
483  hint);
484  else
485  scene =
486  importer.ReadFileFromMemory(&data[0],
487  data.size(),
488  aiProcess_JoinIdenticalVertices | aiProcess_SortByPType | aiProcess_RemoveComponent,
489  hint);
490 
491  if (!scene)
492  {
493  CONSOLE_BRIDGE_logError(
494  "Could not load mesh from \"%s\": %s", resource->getUrl().c_str(), importer.GetErrorString());
495  return std::vector<std::shared_ptr<T>>();
496  }
497 
498  // Assimp enforces Y_UP convention by rotating models with different conventions.
499  // However, that behaviour is confusing and doesn't match the ROS convention
500  // where the Z axis is pointing up.
501  // Hopefully this doesn't undo legit use of the root node transformation...
502  // Note that this is also what RViz does internally.
503  scene->mRootNode->mTransformation = aiMatrix4x4();
504 
505  if (flatten)
506  {
507  // These post processing steps flatten the root node transformation into child nodes,
508  // so they must be delayed until after clearing the root node transform above.
509  importer.ApplyPostProcessing(aiProcess_OptimizeMeshes | aiProcess_OptimizeGraph);
510  }
511  else
512  {
513  importer.ApplyPostProcessing(aiProcess_OptimizeGraph);
514  }
515 
516  return createMeshFromAsset<T>(scene, scale, resource, normals, vertex_colors, material_and_texture);
517 }
